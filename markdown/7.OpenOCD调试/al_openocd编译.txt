WSL2（Linux）：
安装mingw-w64和OpenOCD依赖
sudo apt install mingw-w64

编译安装
./bootstrap && ./configure PKG_CONFIG_PATH=/usr/lib/pkgconfig LDFLAGS="-Wl,-s" --enable-fdti --enable-jlink --enable-ftdi-oscan1 --enable-dummy (--disable-werror) --prefix=/d/debugger/al_openocd_win/build
make -j8 && make install



MSYS2：
安装mingw-w64和OpenOCD依赖
注意：MSYS2安装路径和项目路径不要有空格
pacman -Syu
pacman -Su
pacman -S --needed base-devel mingw-w64-x86_64-toolchain
pacman -S mingw-w64-x86_64-libyaml
pacman -S mingw-w64-x86_64-libusb
pacman -S mingw-w64-x86_64-libftdi
pacman -S libtool
pacman -S autoconf
pacman -S automake
pacman -S texinfo
pacman -S pkg-config
pacman -S make
pacman -S autogen
pacman -S unzip
pacman -S git
使用MSYS2 MinGW X64编译安装:
./bootstrap && ./configure PKG_CONFIG_PATH=/usr/lib/pkgconfig LDFLAGS="-Wl,-s" --enable-fdti --enable-jlink --enable-ftdi-oscan1 --enable-dummy (--disable-werror) --prefix=/d/debugger/al_openocd_win/build
make -j8 && make install  -j多线程执行make



注：
-j：设置多线程执行make编译
LDFLAGS="-Wl,-s"：把选项-s传递给链接器，-s表示从可执行文件中删除符号表和重定位信息，不能再用GDB调试可执行文件
--disable-werror：取消将警告信息当做错误显示
